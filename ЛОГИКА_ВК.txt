0x80
Считывание поданого числа и записывает в X 
X - выбор координаты X
к примеру 3
mov r1, 1
out 0x80, 1 # записать в 0x80, 1
0x81
Считывание поданого числа и записывание в Y
Y - к примеру 2

Пример:

mov r1, 3 ; X pos
mov r2, 2 ; Y pos
out 0x80, r1
out 0x81, r2
mov r1, 0x00 ; Color WHITE
out 0x82, r1
mov r1, 1 ; STATUS: draw
out 0x83, r1
